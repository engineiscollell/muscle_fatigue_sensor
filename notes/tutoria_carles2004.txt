struct OpticalRawFrame {
    uint32_t timestamp_ms;

    uint32_t red;

    float irClean;
    float metric;
};

}
//IMPLEMENTACIO TASKAQC
void taskAcquisition(void *pv) {

    OpticalRawFrame frame{};
    uint8_t state = 0;

    while (true) {

        if (ledRFlag) {

            ledRFlag = false;

            digitalWrite(PIN_LED_RED, HIGH);

            adcStartConversion(state);
        }

        if (ledIRFlag) {

            ledIRFlag = false;

            digitalWrite(PIN_LED_IR, HIGH);

            adcStartConversion(state);
        }

        if (dataReadyFlag) {

            dataReadyFlag = false;

            uint32_t sample =
                adcReadData24();

            switch (state) {

                case 0:
                    frame.red = sample;
                    break;

                case 1:
                    frame.darkRed = sample;
                    break;

                case 2:
                    frame.ir = sample;
                    break;

                case 3:
                    frame.darkIr = sample;

                    frame.timestamp_ms =
                        millis();

                    xQueueSend(
                        g_frameQueue,
                        &frame,
                        0
                    );

                    frame = {};
                    break;
            }

            state =
                (state + 1) % 4;
        }

        vTaskDelay(1);
    }
}
//IMPLEMENTACIÓ TASKPRO
void taskProcessing(void *pv) {

    OpticalRawFrame frame{};
    ProcessedValue value{};

    while (true) {

        if (xQueueReceive(
                g_frameQueue,
                &frame,
                portMAX_DELAY
            ) == pdTRUE) {

            value.redClean =
                (float)frame.red -
                frame.darkRed;

            value.irClean =
                (float)frame.ir -
                frame.darkIr;

            if (value.irClean > 1) {

                value.metric =
                    value.redClean /
                    value.irClean;
            }

            value.timestamp_ms =
                frame.timestamp_ms;

            xQueueSend(
                g_processedQueue,
                &value,
                0
            );
        }
    }
}
//TASKBLE
void taskBLE(void *pv) {

    ProcessedValue value{};

    while (true) {

        if (xQueueReceive(
                g_processedQueue,
                &value,
                portMAX_DELAY
            ) == pdTRUE) {

            uint16_t smo2_x10 =
                (uint16_t)(
                    value.metric * 100.0f
                );

            bleNotifySmO2(
                smo2_x10
            );
        }
    }
}
